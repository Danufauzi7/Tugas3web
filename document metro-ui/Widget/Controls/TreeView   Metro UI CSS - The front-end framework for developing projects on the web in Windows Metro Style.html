    Add leaf on runtime

    Click on leaf, then click button "Add leaf" or click button "Add root"

    • Root
    1. <div class="treeview"
    2.      data-role="treeview"
    3.      id="tree_add_leaf_runtime_example">
    4.     <ul>
    5.         <li><span class="leaf">Root</span></li>
    6.     </ul>
    7. </div>
    9. <script>
    10.     function tree_add_root_runtime_example_click(){
    11.         var tree = $("#tree_add_leaf_runtime_example").data("treeview");
    13.         tree.addLeaf(false, "Root");
    14.     }
    16.     function tree_add_leaf_runtime_example_click(){
    17.         var tree = $("#tree_add_leaf_runtime_example").data("treeview");
    18.         var node = tree.element.find('li.active');
    20.         tree.addLeaf(node, "Leaf item");
    21.     }
    22. </script>
    Add leaf with checkbox and radio

    Click on leaf, then click button "Add leaf" or click button "Add root"

    • Root
    1. <div class="treeview"
    2.      data-role="treeview"
    3.      id="tree_add_leaf_runtime_example2">
    4.     <ul>
    5.         <li><span class="leaf">Root</span></li>
    6.     </ul>
    7. </div>
    9. <script>
    10.     function tree_add_leaf_checkbox_example_click(){
    11.         var tree = $("#tree_add_leaf_runtime_example2").data("treeview");
    12.         var node = tree.element.find('li.active');
    14.         tree.addLeaf(node, "Leaf item", {
    15.             mode: 'checkbox',
    16.             checked: (Math.round(Math.random()) === 0 ? false : true)
    17.         });
    18.     }
    20.     function tree_add_leaf_radio_example_click(){
    21.         var tree = $("#tree_add_leaf_runtime_example2").data("treeview");
    22.         var node = tree.element.find('li.active');
    24.         tree.addLeaf(node, "Leaf item", {
    25.             mode: 'radio',
    26.             name: 'r1',
    27.             checked: (Math.round(Math.random()) === 0 ? false : true)
    28.         });
    29.     }
    30. </script>
